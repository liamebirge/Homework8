List the top 5 Victoria developers by accepted answer percentage.
  1. select top 5
  2. a.OwnerUserId as [User Link],
  3. Count(a.Id) as [Total Answers],
  4. Round(Sum(Case q.AcceptedAnswerId when a.Id then 1 else 0 end) * 100.0 / Count(a.Id), 1) as [Percentage Accepted]
  5. from
  6. Users,
  7. Posts as a
  8. join
  9. Posts as q
  10. on a.ParentId = q.Id
  11. where
  12. q.PostTypeId = 1
  13. and a.PostTypeId = 2
  14. and Users.Location like 'Victoria BC'
  15. group by
  16. a.OwnerUserId
  17. having
  18. Count(a.Id) between 0
  19. and 100
  20. order by
  21. [Percentage Accepted] desc
